# acm Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a fast, single-binary Rust CLI that generates git commit messages using local or remote LLMs, replacing opencommit with progressive diff reduction and streaming output.

**Architecture:** Modular Rust binary with clap CLI, TOML config with env overlay, git shell-outs for diff/commit, two LLM backends (Ollama native + OpenAI-compatible) behind a provider enum, streaming token display via crossterm, and a progressive diff reduction strategy that always makes exactly one LLM call.

**Tech Stack:** Rust (2021 edition), clap 4 (derive), reqwest (rustls + streaming), serde/toml, tokio, crossterm, futures

```rust
use anyhow::Result;
use clap::{Parser, Subcommand};

#[derive(Parser)]
#[command(name = "acm", about = "AI commit message generator")]
struct Cli {
    /// Skip confirmation, commit directly
    #[arg(short = 'y', long)]
    yes: bool,

    /// Provide context hint for the LLM
    #[arg(short, long)]
    context: Option<String>,

    /// Show message but don't commit
    #[arg(long)]
    dry_run: bool,

    /// Internal: used by prepare-commit-msg hook
    #[arg(long, hide = true)]
    hook: Option<String>,

    #[command(subcommand)]
    command: Option<Command>,
}

#[derive(Subcommand)]
enum Command {
    /// Configure acm
    Config {
        #[command(subcommand)]
        action: ConfigAction,
    },
    /// Guided setup: pick provider, model, test connection
    Setup,
    /// Manage git hook
    Hook {
        #[command(subcommand)]
        action: HookAction,
    },
}

#[derive(Subcommand)]
enum ConfigAction {
    /// Set a config value (e.g. model=llama3)
    Set {
        /// Key=value pair
        pair: String,
    },
    /// Show current config
    Show,
}

#[derive(Subcommand)]
enum HookAction {
    /// Install prepare-commit-msg git hook
    Install,
    /// Remove prepare-commit-msg hook
    Uninstall,
}

#[tokio::main]
async fn main() -> Result<()> {
    let cli = Cli::parse();

    match cli.command {
        Some(Command::Config { action }) => match action {
            ConfigAction::Set { pair } => {
                println!("config set: {pair}");
            }
            ConfigAction::Show => {
                println!("config show");
            }
        },
        Some(Command::Setup) => {
            println!("setup");
        }
        Some(Command::Hook { action }) => match action {
            HookAction::Install => println!("hook install"),
            HookAction::Uninstall => println!("hook uninstall"),
        },
        None => {
            println!("generate commit message (not yet implemented)");
        }
    }

    Ok(())
}
```

```rust
/// Path to the config file (~/.config/acm/config.toml)
pub fn config_path() -> PathBuf {
    dirs::config_dir()
        .unwrap_or_else(|| PathBuf::from("."))
        .join("acm")
        .join("config.toml")
}

/// Load config from a specific path. Returns defaults if file missing.
pub fn load_from_path(path: &PathBuf) -> Result<Config> {
    match std::fs::read_to_string(path) {
        Ok(contents) => {
            let cfg: Config = toml::from_str(&contents)
                .with_context(|| format!("invalid config in {}", path.display()))?;
            Ok(cfg)
        }
        Err(e) if e.kind() == std::io::ErrorKind::NotFound => Ok(Config::default()),
        Err(e) => Err(e).with_context(|| format!("reading {}", path.display())),
    }
}

/// Load config from default path with env var overrides applied.
pub fn load() -> Result<Config> {
    let mut cfg = load_from_path(&config_path())?;
    apply_env_overrides(&mut cfg);
    Ok(cfg)
}

fn apply_env_overrides(cfg: &mut Config) {
    apply_env_override(cfg, "ACM_PROVIDER", std::env::var("ACM_PROVIDER").ok().as_deref());
    apply_env_override(cfg, "ACM_MODEL", std::env::var("ACM_MODEL").ok().as_deref());
    apply_env_override(cfg, "ACM_API_URL", std::env::var("ACM_API_URL").ok().as_deref());
    apply_env_override(cfg, "ACM_API_KEY", std::env::var("ACM_API_KEY").ok().as_deref());
    apply_env_override(cfg, "ACM_DIFF_MODE", std::env::var("ACM_DIFF_MODE").ok().as_deref());
    apply_env_override(cfg, "ACM_LANGUAGE", std::env::var("ACM_LANGUAGE").ok().as_deref());
    if let Ok(val) = std::env::var("ACM_MAX_INPUT_TOKENS") {
        if let Ok(n) = val.parse() {
            cfg.max_input_tokens = n;
        }
    }
    if let Ok(val) = std::env::var("ACM_EMOJI") {
        cfg.emoji = val == "true" || val == "1";
    }
    if let Ok(val) = std::env::var("ACM_ONE_LINE") {
        cfg.one_line = val == "true" || val == "1";
    }
}

fn apply_env_override(cfg: &mut Config, key: &str, value: Option<&str>) {
    if let Some(val) = value {
        match key {
            "ACM_PROVIDER" => cfg.provider = val.to_string(),
            "ACM_MODEL" => cfg.model = val.to_string(),
            "ACM_API_URL" => cfg.api_url = Some(val.to_string()),
            "ACM_API_KEY" => cfg.api_key = Some(val.to_string()),
            "ACM_DIFF_MODE" => cfg.diff_mode = val.to_string(),
            "ACM_LANGUAGE" => cfg.language = val.to_string(),
            _ => {}
        }
    }
}

/// Set a key=value in the config file, preserving other values.
pub fn set_value(path: &PathBuf, key: &str, value: &str) -> Result<()> {
    let mut table: toml::Table = match std::fs::read_to_string(path) {
        Ok(contents) => toml::from_str(&contents)
            .with_context(|| format!("parsing {}", path.display()))?,
        Err(e) if e.kind() == std::io::ErrorKind::NotFound => toml::Table::new(),
        Err(e) => return Err(e).with_context(|| format!("reading {}", path.display())),
    };

    // Try to preserve type: booleans and integers
    let toml_value = if value == "true" || value == "false" {
        toml::Value::Boolean(value == "true")
    } else if let Ok(n) = value.parse::<i64>() {
        toml::Value::Integer(n)
    } else {
        toml::Value::String(value.to_string())
    };

    table.insert(key.to_string(), toml_value);

    if let Some(parent) = path.parent() {
        std::fs::create_dir_all(parent)?;
    }
    std::fs::write(path, toml::to_string_pretty(&table)?)?;
    Ok(())
}

/// Format config for display.
pub fn display(cfg: &Config) -> String {
    let mut lines = vec![
        format!("provider = {:?}", cfg.provider),
        format!("model = {:?}", cfg.model),
    ];
    if let Some(url) = &cfg.api_url {
        lines.push(format!("api_url = {:?}", url));
    }
    if cfg.api_key.is_some() {
        lines.push("api_key = (set)".to_string());
    }
    lines.push(format!("max_input_tokens = {}", cfg.max_input_tokens));
    lines.push(format!("emoji = {}", cfg.emoji));
    lines.push(format!("one_line = {}", cfg.one_line));
    lines.push(format!("language = {:?}", cfg.language));
    lines.push(format!("diff_mode = {:?}", cfg.diff_mode));
    lines.join("\n")
}
```

```rust
/// Run a git command in a specific directory.
fn git_in(dir: &Path, args: &[&str]) -> Result<String> {
    let output = Command::new("git")
        .args(args)
        .current_dir(dir)
        .output()
        .context("failed to run git")?;
    if !output.status.success() {
        let stderr = String::from_utf8_lossy(&output.stderr);
        bail!("git {}: {}", args.join(" "), stderr.trim());
    }
    Ok(String::from_utf8_lossy(&output.stdout).to_string())
}

/// List staged files (relative paths). Errors if nothing is staged.
pub fn staged_files(repo: &Path) -> Result<Vec<String>> {
    let output = git_in(repo, &["diff", "--cached", "--name-only"])?;
    let files: Vec<String> = output.lines().filter(|l| !l.is_empty()).map(String::from).collect();
    if files.is_empty() {
        bail!("no staged changes. Stage files with \"git add\" first.");
    }
    Ok(files)
}

/// Get staged diff. `context_lines` sets --unified=N (None = git default of 3).
pub fn staged_diff(repo: &Path, context_lines: Option<u32>) -> Result<String> {
    let ctx_arg;
    let mut args = vec!["diff", "--cached"];
    if let Some(n) = context_lines {
        ctx_arg = format!("--unified={n}");
        args.push(&ctx_arg);
    }
    git_in(repo, &args)
}

/// Get staged diff --stat output.
pub fn staged_stat(repo: &Path) -> Result<String> {
    git_in(repo, &["diff", "--cached", "--stat"])
}

/// Count staged files, insertions, deletions.
pub fn staged_summary(repo: &Path) -> Result<(usize, usize, usize)> {
    let output = git_in(repo, &["diff", "--cached", "--numstat"])?;
    let mut files = 0;
    let mut insertions = 0;
    let mut deletions = 0;
    for line in output.lines() {
        let parts: Vec<&str> = line.split('\t').collect();
        if parts.len() >= 2 {
            files += 1;
            insertions += parts[0].parse::<usize>().unwrap_or(0);
            deletions += parts[1].parse::<usize>().unwrap_or(0);
        }
    }
    Ok((files, insertions, deletions))
}

/// Create a commit with the given message.
pub fn commit(repo: &Path, message: &str) -> Result<()> {
    git_in(repo, &["commit", "-m", message])?;
    Ok(())
}

/// Get the repo root of the current directory.
pub fn repo_root() -> Result<std::path::PathBuf> {
    let output = git(&["rev-parse", "--show-toplevel"])?;
    Ok(PathBuf::from(output.trim()))
}
```

```rust
/// Estimate token count using char/3.2 heuristic. Intentionally conservative for code.
pub fn estimate_tokens(text: &str) -> usize {
    if text.is_empty() {
        return 0;
    }
    (text.len() as f64 / 3.2).ceil() as usize
}

/// Truncate text to fit within max_tokens (estimated). Cuts at last newline before limit.
pub fn truncate_to_tokens(text: &str, max_tokens: usize) -> String {
    if estimate_tokens(text) <= max_tokens {
        return text.to_string();
    }
    let max_bytes = (max_tokens as f64 * 3.2) as usize;
    let truncated = &text[..max_bytes.min(text.len())];
    // Cut at last newline for clean output
    match truncated.rfind('\n') {
        Some(pos) if pos > 0 => truncated[..pos].to_string(),
        _ => truncated.to_string(),
    }
}
```

```rust
/// Select the best diff representation that fits within max_tokens.
pub fn select_diff(
    full: &str,
    compact: &str,
    stat: &str,
    max_tokens: usize,
) -> (String, DiffMode) {
    if token::estimate_tokens(full) <= max_tokens {
        return (full.to_string(), DiffMode::Full);
    }
    if token::estimate_tokens(compact) <= max_tokens {
        return (compact.to_string(), DiffMode::Compact);
    }
    if token::estimate_tokens(stat) <= max_tokens {
        return (stat.to_string(), DiffMode::Stat);
    }
    // Truncate stat to fit
    (token::truncate_to_tokens(stat, max_tokens), DiffMode::Stat)
}

/// Get the diff for a forced mode.
pub fn get_forced_diff(repo: &Path, mode: DiffMode) -> Result<String> {
    match mode {
        DiffMode::Full => git::staged_diff(repo, None),
        DiffMode::Compact => git::staged_diff(repo, Some(0)),
        DiffMode::Stat => git::staged_stat(repo),
    }
}

/// Top-level: fit the staged diff into the token budget.
pub fn fit_diff(repo: &Path, max_tokens: usize, forced_mode: &str) -> Result<(String, DiffMode)> {
    if forced_mode != "auto" {
        let mode = DiffMode::from_str(forced_mode)
            .ok_or_else(|| anyhow::anyhow!("unknown diff mode: {forced_mode}"))?;
        let diff = get_forced_diff(repo, mode)?;
        return Ok((diff, mode));
    }

    let full = git::staged_diff(repo, None)?;
    let compact = git::staged_diff(repo, Some(0))?;
    let stat = git::staged_stat(repo)?;
    Ok(select_diff(&full, &compact, &stat, max_tokens))
}
```

```rust
pub fn detect_scope(files: &[String]) -> Option<String> {
    if files.is_empty() {
        return None;
    }

    // Get directory components only (strip filename)
    let dirs: Vec<Vec<&str>> = files
        .iter()
        .map(|f| {
            let parts: Vec<&str> = f.split('/').collect();
            if parts.len() > 1 {
                parts[..parts.len() - 1].to_vec()
            } else {
                vec![] // root-level file
            }
        })
        .collect();

    // If any file is in root, no common directory
    let min_len = dirs.iter().map(|d| d.len()).min().unwrap_or(0);
    if min_len == 0 {
        return None;
    }

    // Find common directory prefix
    let mut common_depth = 0;
    for i in 0..min_len {
        if dirs.iter().all(|d| d[i] == dirs[0][i]) {
            common_depth = i + 1;
        } else {
            break;
        }
    }

    if common_depth == 0 {
        return None;
    }

    let common = &dirs[0][..common_depth];

    // Strip known prefixes
    let skip = ["src", "pkg", "libs", "apps", "lib", "packages"];
    let start = if common.first().map_or(false, |c| skip.contains(c)) {
        1
    } else {
        0
    };

    let scope_parts = &common[start..];
    if scope_parts.is_empty() {
        None
    } else {
        Some(scope_parts.join("/"))
    }
}
```

```rust
        let body = serde_json::json!({

        });

        if !status.is_success() {
            let body = response.text().await.unwrap_or_default();
            if status.as_u16() == 404 || body.contains("not found") {
                bail!(

                    self.model,

                );
            }
            bail!("Ollama returned {status}: {body}");

        Ok(Box::pin(stream))
    }

/// Parse a single NDJSON line from Ollama's streaming response.
/// Returns the content token, or None if the line has no content.
pub fn parse_ollama_line(line: &str) -> Option<String> {
    let parsed: serde_json::Value = serde_json::from_str(line).ok()?;
    let content = parsed.get("message")?.get("content")?.as_str()?;
    if content.is_empty() {
        None
    } else {
        Some(content.to_string())
    }
}

fn response_to_token_stream(response: reqwest::Response) -> impl futures::Stream<Item = Result<String>> {
    let byte_stream = response.bytes_stream().map(|r| {
        r.map_err(|e| std::io::Error::new(std::io::ErrorKind::Other, e))
    });
    let reader = tokio_util::io::StreamReader::new(byte_stream);
    let lines = tokio::io::BufReader::new(reader).lines();

    futures::stream::unfold(lines, |mut lines| async move {
        loop {
            match lines.next_line().await {
                Ok(Some(line)) => {
                    if let Some(token) = parse_ollama_line(&line) {
                        return Some((Ok(token), lines));
                    }
                    // Empty or unparseable line — skip
                }
                Ok(None) => return None, // stream ended
                Err(e) => return Some((Err(e.into()), lines)),
            }
        }
    })
}
```

```rust
    pub fn new(config: &Config) -> Result<Self> {
        let api_key = config
```

```rust
    pub async fn chat_stream(&self, messages: Vec<Message>) -> Result<TokenStream> {
        let url = format!("{}/v1/chat/completions", self.api_url);
        let body = serde_json::json!({
            "model": self.model,
            "messages": messages,
            "stream": true,
        });

        let response = self
            .client
            .post(&url)
            .header("Authorization", format!("Bearer {}", self.api_key))
            .json(&body)
            .send()
            .await
            .map_err(|e| {
                if e.is_timeout() {
                    anyhow::anyhow!("request timed out after 30s")
                } else if e.is_connect() {
                    anyhow::anyhow!("cannot connect to API at {}", self.api_url)
                } else {
                    e.into()
                }
            })?;

        let status = response.status();
        if !status.is_success() {
            let body = response.text().await.unwrap_or_default();
            bail!("API returned {status}: {body}");
        }

        let stream = response_to_token_stream(response);
        Ok(Box::pin(stream))
    }
}

/// Parse a single SSE line from OpenAI's streaming response.
pub fn parse_openai_line(line: &str) -> Option<String> {
    let data = line.strip_prefix("data: ")?;
    if data == "[DONE]" {
        return None;
    }
    let parsed: serde_json::Value = serde_json::from_str(data).ok()?;
    let content = parsed
        .get("choices")?
        .get(0)?
        .get("delta")?
        .get("content")?
        .as_str()?;
    if content.is_empty() {
        None
    } else {
        Some(content.to_string())
    }
}

fn response_to_token_stream(response: reqwest::Response) -> impl futures::Stream<Item = Result<String>> {
    let byte_stream = response.bytes_stream().map(|r| {
        r.map_err(|e| std::io::Error::new(std::io::ErrorKind::Other, e))
    });
    let reader = tokio_util::io::StreamReader::new(byte_stream);
    let lines = tokio::io::BufReader::new(reader).lines();

    futures::stream::unfold(lines, |mut lines| async move {
        loop {
            match lines.next_line().await {
                Ok(Some(line)) => {
                    if let Some(token) = parse_openai_line(&line) {
                        return Some((Ok(token), lines));
                    }
                }
                Ok(None) => return None,
                Err(e) => return Some((Err(e.into()), lines)),
            }
        }
    })
}
```

```rust
/// Stream tokens from the LLM to stdout, collecting the full message.
pub async fn stream_message(stream: &mut TokenStream) -> Result<String> {
    let mut message = String::new();
    let mut stdout = io::stdout();

    while let Some(token) = stream.next().await {
        let token = token?;
        message.push_str(&token);
        print!("{token}");
        stdout.flush()?;
    }
    println!();

    Ok(message)
}

/// Show the action prompt and wait for a single keypress.
pub fn prompt_action() -> Result<Action> {
    eprintln!();
    eprintln!("  [y] commit  [e] edit  [r] regenerate  [n] cancel");

    enable_raw_mode()?;
    let action = loop {
        if let Event::Key(key) = read()? {
            let action = match key.code {
                KeyCode::Char('y') | KeyCode::Enter => Some(Action::Commit),
                KeyCode::Char('e') => Some(Action::Edit),
                KeyCode::Char('r') => Some(Action::Regenerate),
                KeyCode::Char('n') | KeyCode::Esc => Some(Action::Cancel),
                KeyCode::Char('c')
                    if key.modifiers.contains(crossterm::event::KeyModifiers::CONTROL) =>
                {
                    Some(Action::Cancel)
                }
                _ => None,
            };
            if let Some(a) = action {
                break a;
            }
        }
    };
    disable_raw_mode()?;

    Ok(action)
}

/// Open $EDITOR with the message pre-filled for editing.
pub fn edit_message(message: &str) -> Result<String> {
    let editor = std::env::var("EDITOR").unwrap_or_else(|_| "vi".into());
    let tmp = tempfile::NamedTempFile::new()?;
    std::fs::write(tmp.path(), message)?;

    let status = std::process::Command::new(&editor)
        .arg(tmp.path())
        .status()?;

    if !status.success() {
        anyhow::bail!("editor exited with {status}");
    }

    let edited = std::fs::read_to_string(tmp.path())?;
    Ok(edited.trim().to_string())
```

```rust
    Setup,
    /// Manage git hook
    Hook {
        #[command(subcommand)]

#[derive(Subcommand)]
enum ConfigAction {
    /// Set a config value (e.g. model=llama3)
    Set { pair: String },
    /// Show current config
    Show,
}

#[derive(Subcommand)]
enum HookAction {
    /// Install prepare-commit-msg git hook
    Install,
    /// Remove prepare-commit-msg hook
    Uninstall,
}

#[tokio::main]
async fn main() -> Result<()> {
    let cli = Cli::parse();

    match cli.command {
        Some(Command::Config { action }) => match action {
            ConfigAction::Set { pair } => {
                let (key, value) = pair
                    .split_once('=')
                    .context("expected key=value format")?;
                config::set_value(&config::config_path(), key.trim(), value.trim())?;
                eprintln!("set {} = {}", key.trim(), value.trim());
            }
            ConfigAction::Show => {
                let cfg = config::load()?;
                println!("{}", config::display(&cfg));
            }
        },
        Some(Command::Setup) => {
            setup::run().await?;
        }
        Some(Command::Hook { action }) => match action {
            HookAction::Install => hook::install()?,
            HookAction::Uninstall => hook::uninstall()?,
        },
        None => {
            generate(&cli).await?;
        }
    }

    Ok(())
}

async fn generate(cli: &Cli) -> Result<()> {
    let cfg = config::load()?;
    let repo = git::repo_root()?;

    let files = git::staged_files(&repo)?;
    let (file_count, ins, del) = git::staged_summary(&repo)?;

    if !cli.dry_run && cli.hook.is_none() {
        ui::print_summary(file_count, ins, del);
    }

    let detected_scope = scope::detect_scope(&files);
    let system_prompt = prompt::build_system_prompt(&cfg, detected_scope.as_deref());

    let (diff_text, _diff_mode) = diff::fit_diff(&repo, cfg.max_input_tokens, &cfg.diff_mode)?;

    let mut user_content = diff_text;
    if let Some(ctx) = &cli.context {
        user_content = format!("Context: {ctx}\n\n{user_content}");
    }

    let messages = vec![
        llm::Message { role: llm::Role::System, content: system_prompt },
        llm::Message { role: llm::Role::User, content: user_content },
    ];

    let provider = llm::Provider::from_config(&cfg)?;

    loop {
        let mut stream = provider.chat_stream(messages.clone()).await?;
        let message = ui::stream_message(&mut stream).await?;

        // Hook mode: write to commit message file and exit
        if let Some(ref hook_path) = cli.hook {
            std::fs::write(hook_path, &message)?;
            return Ok(());
        }

        // Dry-run mode: just show the message and exit
        if cli.dry_run {
            return Ok(());
        }

        // Auto-confirm mode
        if cli.yes {
            git::commit(&repo, &message)?;
            return Ok(());
        }

        // Interactive mode
        match ui::prompt_action()? {
            ui::Action::Commit => {
                git::commit(&repo, &message)?;
                return Ok(());
            }
            ui::Action::Edit => {
                let edited = ui::edit_message(&message)?;
                if !edited.is_empty() {
                    git::commit(&repo, &edited)?;
                }
                return Ok(());
            }
            ui::Action::Regenerate => {
                eprintln!();
                continue; // loop back to generate again
            }
            ui::Action::Cancel => {
                return Ok(());
            }
        }
    }
}
```

```rust
async fn test_connection(provider: &str, api_url: &str, api_key: Option<&str>) -> bool {
    let client = reqwest::Client::builder()
        .timeout(std::time::Duration::from_secs(5))
```
